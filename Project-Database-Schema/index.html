Architectural Overview

This is a production-ready, highly normalized PostgreSQL schema designed for scalable Warehouse Management Systems (WMS). It strictly adheres to 3NF standards to eliminate data redundancy while maintaining audit tracks for complex supply chains.

  • Location Granularity: Models multi-warehouse networks down to precisely localized bin geometries (Aisle/Rack/Shelf). Enables A* snake-routing algorithms for pickers.
  • Ledger Paradigm: Tracks inventory state through immutable Inventory_Transactions ensuring perfect referential and historical integrity over Stock_Levels.
  • Analytics Ready: Uses an optimized MATERIALIZED VIEW for BI platforms like Power BI, aggregating daily KPI metrics (capacity utilization, reorder threshold triggers) natively within the database to ensure sub-second rendering.
init_warehouse_schema.sql 
-- Create custom enum for zone types
CREATE TYPE zone_type_enum AS ENUM ('AMBIENT', 'COLD_STORAGE', 'HAZARDOUS', 'HIGH_VALUE');

CREATE TABLE Warehouses (
    warehouse_id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE,
    location_code VARCHAR(10) NOT NULL UNIQUE,
    address TEXT NOT NULL,
    total_capacity_cbm NUMERIC(10, 2) NOT NULL, -- Total Cubic Meters available
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Storage_Zones (
    zone_id SERIAL PRIMARY KEY,
    warehouse_id INT NOT NULL REFERENCES Warehouses(warehouse_id) ON DELETE CASCADE,
    zone_code VARCHAR(20) NOT NULL,
    zone_type zone_type_enum NOT NULL DEFAULT 'AMBIENT',
    temperature_range VARCHAR(50), -- E.g., "-18C to -20C"
    capacity_cbm NUMERIC(10, 2) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(warehouse_id, zone_code)
);

CREATE TABLE Bin_Locations (
    bin_id SERIAL PRIMARY KEY,
    zone_id INT NOT NULL REFERENCES Storage_Zones(zone_id) ON DELETE CASCADE,
    aisle VARCHAR(10) NOT NULL,
    rack VARCHAR(10) NOT NULL,
    shelf VARCHAR(10) NOT NULL,
    bin_position VARCHAR(10) NOT NULL,
    barcode VARCHAR(50) NOT NULL UNIQUE,
    max_weight_kg NUMERIC(8, 2),
    volume_cbm NUMERIC(8, 4),
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(zone_id, aisle, rack, shelf, bin_position)
);

CREATE TYPE uom_enum AS ENUM ('EACH', 'BOX', 'PALLET', 'KG', 'LITER');

CREATE TABLE Products (
    product_id SERIAL PRIMARY KEY,
    sku VARCHAR(50) NOT NULL UNIQUE,
    name VARCHAR(255) NOT NULL,
    category VARCHAR(100) NOT NULL,
    description TEXT,
    uom uom_enum NOT NULL DEFAULT 'EACH',
    unit_weight_kg NUMERIC(8, 2) NOT NULL,
    unit_volume_cbm NUMERIC(8, 4) NOT NULL,
    reorder_point INT NOT NULL DEFAULT 0,
    requires_cold_chain BOOLEAN DEFAULT FALSE,
    is_hazardous BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Tracks real-time precise location of inventory
CREATE TABLE Stock_Levels (
    stock_id SERIAL PRIMARY KEY,
    product_id INT NOT NULL REFERENCES Products(product_id),
    bin_id INT NOT NULL REFERENCES Bin_Locations(bin_id),
    quantity INT NOT NULL CHECK (quantity >= 0),
    batch_number VARCHAR(100),
    expiry_date DATE,
    last_counted_at TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(product_id, bin_id, batch_number) -- A specific batch of a product in a specific bin
);

CREATE TYPE transaction_type_enum AS ENUM ('RECEIPT', 'PICK', 'PUTAWAY', 'ADJUSTMENT_UP', 'ADJUSTMENT_DOWN', 'TRANSFER');

-- Immutable ledger of all inventory movements
CREATE TABLE Inventory_Transactions (
    transaction_id SERIAL PRIMARY KEY,
    product_id INT NOT NULL REFERENCES Products(product_id),
    from_bin_id INT REFERENCES Bin_Locations(bin_id), -- Null if external receipt
    to_bin_id INT REFERENCES Bin_Locations(bin_id),   -- Null if external shipment
    transaction_type transaction_type_enum NOT NULL,
    quantity INT NOT NULL CHECK (quantity > 0),
    reference_id VARCHAR(100), -- E.g., PO Number or SO Number
    performed_by INT, -- References Employees table (defined later)
    transaction_date TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TYPE order_status_enum AS ENUM ('DRAFT', 'PENDING', 'PROCESSING', 'SHIPPED', 'DELIVERED', 'CANCELLED');

-- Supplier side (Inbound)
CREATE TABLE Suppliers (
    supplier_id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    contact_email VARCHAR(255),
    lead_time_days INT NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Purchase_Orders (
    po_id SERIAL PRIMARY KEY,
    po_number VARCHAR(50) NOT NULL UNIQUE,
    supplier_id INT NOT NULL REFERENCES Suppliers(supplier_id),
    destination_warehouse_id INT NOT NULL REFERENCES Warehouses(warehouse_id),
    status order_status_enum NOT NULL DEFAULT 'PENDING',
    expected_delivery_date DATE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE PO_Lines (
    po_line_id SERIAL PRIMARY KEY,
    po_id INT NOT NULL REFERENCES Purchase_Orders(po_id) ON DELETE CASCADE,
    product_id INT NOT NULL REFERENCES Products(product_id),
    quantity_ordered INT NOT NULL CHECK (quantity_ordered > 0),
    quantity_received INT NOT NULL DEFAULT 0 CHECK (quantity_received >= 0),
    unit_cost NUMERIC(10, 2) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Customer side (Outbound)
CREATE TABLE Sales_Orders (
    so_id SERIAL PRIMARY KEY,
    so_number VARCHAR(50) NOT NULL UNIQUE,
    customer_name VARCHAR(255) NOT NULL,
    shipping_address TEXT NOT NULL,
    origin_warehouse_id INT NOT NULL REFERENCES Warehouses(warehouse_id),
    status order_status_enum NOT NULL DEFAULT 'PENDING',
    priority_level INT DEFAULT 3, -- 1 is highest
    required_ship_date DATE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE SO_Lines (
    so_line_id SERIAL PRIMARY KEY,
    so_id INT NOT NULL REFERENCES Sales_Orders(so_id) ON DELETE CASCADE,
    product_id INT NOT NULL REFERENCES Products(product_id),
    quantity_ordered INT NOT NULL CHECK (quantity_ordered > 0),
    quantity_picked INT NOT NULL DEFAULT 0 CHECK (quantity_picked >= 0),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TYPE role_enum AS ENUM ('MANAGER', 'SUPERVISOR', 'PICKER', 'RECEIVER', 'FORKLIFT_OP');
CREATE TYPE task_status_enum AS ENUM ('PENDING', 'IN_PROGRESS', 'COMPLETED', 'BLOCKED');
CREATE TYPE task_type_enum AS ENUM ('PICK', 'PUTAWAY', 'CYCLE_COUNT', 'REPLENISHMENT');

CREATE TABLE Employees (
    employee_id SERIAL PRIMARY KEY,
    first_name VARCHAR(100) NOT NULL,
    last_name VARCHAR(100) NOT NULL,
    role role_enum NOT NULL,
    primary_warehouse_id INT REFERENCES Warehouses(warehouse_id),
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Add foreign key back to Inventory_Transactions now that Employees exist
ALTER TABLE Inventory_Transactions 
ADD CONSTRAINT fk_performed_by 
FOREIGN KEY (performed_by) REFERENCES Employees(employee_id);

CREATE TABLE Task_Assignments (
    task_id SERIAL PRIMARY KEY,
    task_type task_type_enum NOT NULL,
    status task_status_enum NOT NULL DEFAULT 'PENDING',
    assigned_to INT REFERENCES Employees(employee_id),
    related_order_id INT, -- Can link to PO or SO depending on context
    target_bin_id INT REFERENCES Bin_Locations(bin_id),
    product_id INT REFERENCES Products(product_id),
    required_quantity INT,
    completed_at TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Carriers (
    carrier_id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE,
    scac_code VARCHAR(10), -- Standard Carrier Alpha Code
    contact_number VARCHAR(50),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Shipments (
    shipment_id SERIAL PRIMARY KEY,
    so_id INT NOT NULL REFERENCES Sales_Orders(so_id),
    carrier_id INT NOT NULL REFERENCES Carriers(carrier_id),
    tracking_number VARCHAR(100) UNIQUE,
    total_weight_kg NUMERIC(10, 2),
    estimated_delivery DATE,
    actual_freight_cost NUMERIC(10, 2),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Materialized View for Daily Inventory Turnover and Capacity metrics
CREATE MATERIALIZED VIEW mv_warehouse_analytics AS
SELECT 
    DATE(CURRENT_DATE) as report_date,
    w.warehouse_id,
    w.name AS warehouse_name,
    w.total_capacity_cbm,
    SUM(sl.quantity * p.unit_volume_cbm) AS current_utilized_cbm,
    ROUND((SUM(sl.quantity * p.unit_volume_cbm) / w.total_capacity_cbm) * 100, 2) AS utilization_percentage,
    COUNT(DISTINCT sl.product_id) AS distinct_skus_in_stock,
    SUM(CASE WHEN sl.quantity <= p.reorder_point THEN 1 ELSE 0 END) AS skus_below_reorder_point
FROM 
    Warehouses w
JOIN 
    Storage_Zones sz ON w.warehouse_id = sz.warehouse_id
JOIN 
    Bin_Locations bl ON sz.zone_id = bl.zone_id
LEFT JOIN 
    Stock_Levels sl ON bl.bin_id = sl.bin_id
LEFT JOIN 
    Products p ON sl.product_id = p.product_id
GROUP BY 
    w.warehouse_id, w.name, w.total_capacity_cbm;

-- Create an index on the materialized view for fast BI querying
CREATE UNIQUE INDEX idx_mv_analytics_wh ON mv_warehouse_analytics(report_date, warehouse_id);
